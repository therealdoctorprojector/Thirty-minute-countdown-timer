
// Target: ATtiny85
// Board: Bare ATtiny85, 1MHz internal oscillator
// Programmer: ICSP via USB
//
// OLED details
// Type SSD1306
// Voltage range 3 to 5V DC
// Display 128x64 pixels
//
// Connections:
// ATTiny85 pin 5 (PB0/MOSI/SDA) to OLED SDA
// ATTiny85 pin 7 (PB2/SCK/SCL) to OLED SCL
// ATTiny85 pin 2 (PCINT3/CLKI/OC1B/ADC3) to (+ve) on beeper device

#include <TinyWireM.h>
#define TINY4KOLED_QUICK_BEGIN
#include <Tiny4kOLED.h>

void setup() {

  // Initialize OLED screen
  oled.begin(128, 64, sizeof(tiny4koled_init_128x64br), tiny4koled_init_128x64br);
  oled.enableChargePump(); // The default is off, but most boards need this.
  oled.setRotation(1);     // The default orientation is not the most commonly used.

  // Two fonts are supplied with this library, FONT8X16 and FONT6X8
  oled.setFontX2Smooth(FONT8X16);

  // To clear all the memory
  oled.clear();
  oled.on();

  // Set up the peeper
  pinMode(3, OUTPUT);
  digitalWrite(3, LOW);

}
